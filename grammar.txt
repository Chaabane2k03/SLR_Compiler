<programme> ::= PROGRAM <nom_programme> ';' <bloc> '.'

<bloc> ::= VAR <declaration_variable> BEGIN <instruction> END

<declaration_variable> ::= <nom_variable> ':' <type> ';' | <declaration_variable> <nom_variable> ':' <type> ';'

<instruction> ::= <assignation> | <condition> | <boucle> | <procedure> | <fonction> 

<assignation> ::= <nom_variable> ':=' <expression>

<condition> ::= IF <expression> THEN <instruction> [ELSE <instruction>]

<boucle> ::= WHILE <expression> DO <instruction> | FOR <nom_variable> ':=' <expression> TO <expression> DO <instruction>

<procedure> ::= PROCEDURE <nom_procedure> ';' <bloc> END ';'

<fonction> ::= FUNCTION <nom_fonction> ':' <type> ';' <bloc> END ';'

<expression> ::= <expression_arithmetique> | <expression_logique> | <nom_variable>

<expression_arithmetique> ::= <expression> ('+' | '-' | '*' | '/') <expression> | <entier>

<expression_logique> ::= <expression> ('AND' | 'OR' | 'NOT') <expression>

<type> ::= INTEGER | REAL | BOOLEAN
